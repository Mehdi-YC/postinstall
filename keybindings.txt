# apps
open bfowser : super + b
open terminal : super + return
open file manager : super + f
dmenu/rofi : super + o

# windows :
go to window in X direction : super + X

# workspaces :
go to the next workspace : super + ctrl + right
go to the prev workspace : super + ctrl + left
go th X workspace : super (+ ctrl) + x
send to the next workspace and follow : super + shift + right
send to the prev workspace and follow : super + shift + left

# screens :
send to the other screen : super + o
send to the left screen : super + shift + ctrl + right
send to the right screen : super + shift + ctrl + left
